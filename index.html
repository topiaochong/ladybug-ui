    # 快速开始

    # 安装组件库

    npm i ladybug-ui

    # 引用组件库

    在 main.js 中引用组件库

    //全部引入
import "ladybug-ui/dist/css/index.css";
import LadyBugUI from "ladybug-ui";
Vue.use(LadyBugUI);

//按需引入
import "ladybug-ui/dist/css/demo.css";
import { Demo } from "ladybug-ui";
Vue.use(Demo);
